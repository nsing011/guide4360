// Prisma schema for MongoDB
// Copy this entire content and replace the contents of prisma/schema.prisma on your other device

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mongodb"
  url      = env("DATABASE_URL")
}

model User {
  id             String          @id @default(auto()) @map("_id") @db.ObjectId
  username       String          @unique
  password       String
  createdAt      DateTime        @default(now())
  tasks          Task[]
  automatedTasks AutomatedTask[]
  pipelines      Pipeline[]      @relation("UserPipelines")
  pipelineMonitoring PipelineMonitoring[] @relation("UserPipelineMonitoring")
}

model Task {
  id          String   @id @default(auto()) @map("_id") @db.ObjectId
  retailer    String
  day         String
  fileCount   Int
  xlsxCount   Int      @default(0)
  csvCount    Int      @default(0)
  txtCount    Int      @default(0)
  mailCount   Int      @default(0)
  loadType    String   // "Direct load" or "Indirect load"
  
  // Direct Load fields
  istTime     String?
  estTime     String?
  sqlQuery    String?  @db.String
  
  // Indirect Load fields
  indirectLoadSource String? // "retailer portal" or "retailer mail"
  
  // Retailer Portal fields
  websiteLink String?
  portalUsername String?
  portalPassword String?
  
  // Retailer Mail fields
  mailFolder  String?
  mailId      String?
  
  // Legacy fields
  link        String   @default("")
  username    String   @default("")
  password    String   @default("")
  
  files       TaskFile[]
  completed   Boolean  @default(false)
  completedBy String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  ktRecordingLink    String?
  documentationLink  String?
  instructions       String?
  
  userId      String   @db.ObjectId
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  @@index([userId])
  @@index([completed])
}

model TaskFile {
  id            String @id @default(auto()) @map("_id") @db.ObjectId
  downloadName  String
  requiredName  String
  taskId        String @db.ObjectId
  task          Task   @relation(fields: [taskId], references: [id], onDelete: Cascade)
  
  @@index([taskId])
}

model AutomatedTask {
  id          String   @id @default(auto()) @map("_id") @db.ObjectId
  name        String
  description String?
  taskType    String   // e.g., "retailer_data", "inventory_update", "sales_report"
  isActive    Boolean  @default(true)
  lastRun     DateTime?
  nextRun     DateTime?
  uploadedFile String? // Store file path or filename
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  userId      String   @db.ObjectId
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  @@index([userId])
}

model Pipeline {
  id           String   @id @default(auto()) @map("_id") @db.ObjectId
  triggerName  String
  description  String?
  shift        String   // "A", "B", or "C"
  isActive     Boolean  @default(true)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
  userId       String   @db.ObjectId
  user         User     @relation("UserPipelines", fields: [userId], references: [id], onDelete: Cascade)
  
  @@index([userId])
}

model PipelineMonitoring {
  id              String   @id @default(auto()) @map("_id") @db.ObjectId
  date            DateTime @default(now())
  shift           String   // "A", "B", or "C"
  triggerName     String
  runId           String
  status          String   // "SUCCESS", "FAILED", "RUNNING", "SKIPPED"
  monitoredBy     String   // Username of who monitored
  reRunId         String?
  incNumber       String?  // INC ticket number
  currentStatus   String?  // Status after re-run: "RESOLVED", "FAILED_AGAIN", "PENDING"
  resolvedBy      String?  // "L1", "L2", "OPS"
  workingTeam     String?  // "L1_TEAM", "L2_TEAM", "OPS_TEAM", "L1_WORKING", "L2_WORKING", "OPS_WORKING"
  comments        String?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  userId          String   @db.ObjectId
  user            User     @relation("UserPipelineMonitoring", fields: [userId], references: [id], onDelete: Cascade)
  
  @@index([userId])
  @@index([date])
  @@index([shift])
}
